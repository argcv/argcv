55 // Clean returns the shortest path name equivalent to path
56 // by purely lexical processing. It applies the following rules
57 // iteratively until no further processing can be done:
58 //
59 // 1. Replace multiple slashes with a single slash.
60 // 2. Eliminate each . path name element (the current directory).
61 // 3. Eliminate each inner .. path name element (the parent directory)
62 // along with the non-.. element that precedes it.
63 // 4. Eliminate .. elements that begin a rooted path:
64 // that is, replace "/.." by "/" at the beginning of a path.
65 //
66 // The returned path ends in a slash only if it is the root "/".
67 //
68 // If the result of this process is an empty string, Clean
69 // returns the string ".".
70 inline string CleanPath(string_view in) noexcept {
71  if (in == "") {
72  return ".";
73  }
74 
75  // Invariants:
76  // reading from path; r is index of next byte to process.
77  // writing to buf; w is index of next byte to write.
78  // dotdot is index in buf where .. must stop, either because
79  // it is the leading slash or it is a leading ../../.. prefix.
80  bool rooted = in[0] == '/';
81  internal::PathBuff out(rooted ? "/" : "");
82  size_t r = 0;
83  size_t dotdot = 0;
84  size_t n = in.length();
85  if (rooted) {
86  r++;
87  dotdot++;
88  }
89  while (r < n) {
90  // current character
91  char c = in[r];
92  switch (c) {
93  case '/':
94  // empty path element
95  r++;
96  break;
97  case '.':
98  if (r + 1 <= n) {
99  if (r + 1 == n || in[r + 1] == '/') {
100  // . element
101  r++;
102  break; // break switch(c)
103  } else if (r + 2 <= n && in[r + 1] == '.' &&
104  (r + 2 == n || in[r + 2] == '/')) {
105  // .. element: remove to last /
106  r += 2;
107  if (out.W() > dotdot) {
108  out.Backspace();
109  while (out.W() > dotdot && out.Index(out.W()) != '/') {
110  out.Backspace();
111  }
112  } else if (!rooted) {
113  if (out.W() > 0) {
114  out.Append('/');
115  }
116  out.Append('.');
117  out.Append('.');
118  dotdot = out.W();
119  }
120  break;
121  } else {
122  // NO break, to default
123  }
124  } else {
125  // NO break, to default
126  }
127  // if not eligible, no break, execute default
128  default:
129  // real path element.
130  // add slash if needed
131  if ((rooted && out.W() != 1) || (!rooted && out.W() != 0)) {
132  out.Append('/');
133  }
134  // copy element
135  while (r < n && in[r] != '/') {
136  out.Append(in[r]);
137  r++;
138  }
139  }
140  }
141 
142  if (out.W() == 0) {
143  return ".";
144  }
145 
146  return out.ToString();
147 }

189 /// Dirname returns all but the last element of path, typically the path's
190 /// directory.
191 ///
192 /// After dropping the final element using Split, the path is Cleaned
193 /// and trailing slashes are removed. If the path is empty, Dirname returns ".".
194 /// If the path consists entirely of slashes followed by non-slash bytes,
195 /// Dirname returns a single slash. In any other case, the returned path does
196 /// not end in a slash.

201  return CleanPath(dir);
