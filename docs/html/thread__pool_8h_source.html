43  // Send a callback function, expect nothing to return
44  void Send(const std::function<void()>& task) noexcept {
45  std::lock_guard<std::mutex> lock(mu_);
46  // Add works to the callbacks list
47  tasks_.emplace(task);
48  // Increase pool size or notify as needed
49  if (threads_waiting_ == 0 &&
50  ((max_parallelism_ == 0 || (nthreads_ < max_parallelism_)))) {
51  nthreads_++;
52  new DynamicThread(this);
53  } else {
54  cv_.notify_one();
55  }
56  // Also use this chance to harvest dead threads
57  if (!zombie_threads_.empty()) {
58  CleanZombieThreads(&zombie_threads_);
59  }
60  }

62  // add new work item to the pool
63  template <class F, class... Args>
64  auto Enqueue(F&& f, Args&&... args) noexcept // NOLINT(build/c++11)
65  -> std::future<typename std::result_of<F(Args...)>::type> {
66  using return_type = typename std::result_of<F(Args...)>::type;
67 
68  auto task = std::make_shared<std::packaged_task<return_type()>>(
69  std::bind(std::forward<F>(f),
70  std::forward<Args>(args)...)); // NOLINT(build/c++11)
71 
72  std::future<return_type> res = task->get_future();
73 
74  Send([task]() { (*task)(); });
75  return res;
76  }

123  void ThreadFunc() noexcept {
124  for (;;) {
125  // Wait until work is available or we are shutting down.
126  std::unique_lock<std::mutex> lock(mu_);
127  if (!shutdown_ && tasks_.empty()) {
128  // If there are too many threads waiting, then quit this thread
129  if (threads_waiting_ >= reserve_threads_) {
130  break;
131  }
132  threads_waiting_++;
133  cv_.wait(lock);
134  threads_waiting_--;
135  }
136  // Drain callbacks before considering shutdown to ensure all work
137  // gets completed.
138  if (!tasks_.empty()) {
139  auto task = tasks_.front();
140  tasks_.pop();
141  lock.unlock();
142  task();
143  } else if (shutdown_) {
144  break;
145  }
146  }
147  }
