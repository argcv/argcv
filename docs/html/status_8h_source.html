37 /**
38  * \brief A generic return type
39  *
40  * There are a few ways to return a Status with error message.
41  * In this project, we are supposed to create a global type, contains
42  * Successful information with error message instead of throwing a exception.
43  */
44 class Status {
45  public:
46  Status() noexcept : state_(nullptr) {}
47  ~Status() { delete[] state_; }
48 
49  Status(const Status& rhs) noexcept {
50  state_ = (rhs.state_ == nullptr) ? nullptr : CopyState(rhs.state_);
51  }

53  /// copy
54  Status& operator=(const Status& rhs) noexcept {
55  /// The following condition catches both aliasing (when this == &rhs),
56  /// and the common case where both rhs and *this are ok.
57  if (state_ != rhs.state_) {
58  delete[] state_;
59  state_ = (rhs.state_ == nullptr) ? nullptr : CopyState(rhs.state_);
60  }
61  return *this;
62  }

66  std::swap(state_, rhs.state_);

102  /// Return a string representation of this status suitable for printing.
103  /// Returns the string "OK" for success.
104  std::string ToString() const noexcept {
105  if (state_ == nullptr) {
106  return "OK";
107  } else {
108  char tmp[30];
109  const char* type;
110  int icode = static_cast<int>(code());
111  auto it = code_desc_.find(icode);
112  if (it != code_desc_.end()) {
113  snprintf(tmp, sizeof(tmp), "Error %s (%d): ", it->second.c_str(),
114  icode);
115  } else {
116  snprintf(tmp, sizeof(tmp), "Unknown Error (%d): ", icode);
117  }
118  type = tmp;
119  std::string result(type);
120  uint32_t length;
121  memcpy(&length, state_, sizeof(length));
122  result.append(state_ + 5, length);
123  return result;
124  }

127  template <typename... Args>
128  Status& AppendToMessage(Args... args) noexcept {
129  ECode new_code = code();
130  if (ok()) {
131  new_code = ::argcv::error::kUnknown;
132  }
133  std::string msg = absl::StrCat(error_message(), "\n\t", args...);
134  const uint32_t size = msg.size();
135  char* result = new char[size + 5];
136  memcpy(result, &size, sizeof(size));
137  result[4] = static_cast<char>(new_code);
138  memcpy(result + 5, msg.data(), size);
139  std::swap(state_, result);
140  return *this;
141  }
